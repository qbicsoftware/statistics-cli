1   package life.qbic.io.writer;

3   import org.apache.logging.log4j.LogManager;
4   import org.apache.logging.log4j.Logger;
5   import org.yaml.snakeyaml.DumperOptions;
6   import org.yaml.snakeyaml.Yaml;
7   import submodule.data.MainConfig;
8   
9   import java.io.FileWriter;
10  import java.io.IOException;
11  
12  /**
13   * @author fhanssen
14   * Writes given MainConfig to file. DON'T change the DumperOptions, unless you are absolutely sure of what you are doing.
15   * This has to be synced with the Reader in the statistics portlet.
16   */
17  public final class YAMLWriter {
18  
19      private static final Logger logger = LogManager.getLogger(YAMLWriter.class);
20  
21      public static void writeToFile(String outputFile, MainConfig config){
22          DumperOptions options = new DumperOptions();
23          options.setDefaultFlowStyle(DumperOptions.FlowStyle.BLOCK);
24          options.setPrettyFlow(true);
25          Yaml yaml = new Yaml(options);
26  
27          try {
28              FileWriter writer = new FileWriter(outputFile);
29              yaml.dump(config, writer);
30              logger.info("Charts were successfully written to config");
31          }catch(IOException e){
32              e.printStackTrace();
33              logger.error("Config file could not be written: " + e.getMessage());
34          }
35      }
36  }
