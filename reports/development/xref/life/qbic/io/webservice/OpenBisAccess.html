8   /**
9    * @author fhanssen
10   * Creates access to openBis by generatin the sessionToken.
11   */
12  public class OpenBisAccess {
13  
14      private static final Logger logger = LogManager.getLogger(OpenBisAccess.class);
15  
16      private final String URL;
17      private final int TIMEOUT = 100000;
18      private final IApplicationServerApi v3;
19      private final String user;
20      private final String password;
21  
22      private String sessionToken = "";
23  
24      public OpenBisAccess(String url, String user, String password) {
25          this.user = user;
26          this.password = password;
27          this.URL = url +  IApplicationServerApi.SERVICE_URL;
28  
29          v3 = HttpInvokerUtils.createServiceStub(IApplicationServerApi.class, URL, TIMEOUT);
30          login();
31          logger.info("Login to OpenBis");
32      }
33  
34      private void login() {
35          sessionToken = (v3.login(user, password));
36      }
37  
38      public String getSessionToken() {
39          return sessionToken;
40      }
41  
42      public IApplicationServerApi getV3() {
43          return v3;
44      }
45  
46      public void logout() {
47          v3.logout(sessionToken);
48      }
49  }
