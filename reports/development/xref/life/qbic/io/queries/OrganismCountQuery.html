30  /**
31   * @author fhanssen
32   * This Query class, accesses OpenBis and retrieves all samples. They are then counted once by the superkingdom they belong to,
33   * once by genus and once by species. Species, which occupy large parts(> DOMAIN_THRESHOLD, currently 25%) of a superkingdom,
34   * are shown in the superkingdom resolution.
35   */
36  public class OrganismCountQuery extends AQuery {
37  
38      private static final Logger logger = LogManager.getLogger(OrganismCountQuery.class);
39  
40      private final IApplicationServerApi v3;
41      private final String sessionToken;
42      private final String ncbiTaxonomyRestUrl;
43      private final double DOMAIN_THRESHOLD; //DOMAIN_THRESHOLD for when a fraction of species in a kingdom counts as large
44  
45      private  SearchResult<Sample> searchResult;
46  
47      private final List<String> largeSpecies = new ArrayList<>(); //Species >= largeThreshold share
48      private final Set<String> largeDomains = new HashSet<>();
49  
50      private final Map<String, String> vocabularyMap = new HashMap<>(); //511145=Escherichia coli strain K12 MG1655 [NCBI ID = Name]
51  
52      private final Map<String, Integer> organismCountMap = new HashMap<>(); //103690=26 [NCBI ID = count]
53      private final Map<String, String> organismDomainMap = new HashMap<>(); //103690=Bacteria [NCBI ID = domain]
54      private final Map<String, String> organismGenusMap = new HashMap<>(); //103690=Nostoc [NCBI ID = Genus]
55  
56      //Result maps
57      private final Map<String, Object> domainCountMap = new HashMap<>(); // Eukaryota=2000, Bacteria = 200, Homo Sapiens = 5000,...
58      private final Map<String, Map<String, Object>> genusCountMaps = new HashMap<>(); //Eukaryota, Map[Mus = 26]
59      private final Map<String, Map<String, Object>> speciesCountMaps = new HashMap<>(); //Eukaryota, Map[Mus musculus = 23]
60  
61      private final Map<String, Object> organismNameGenusMap = new HashMap<>(); //Cavia aperea=Cavia [Species name = Genus]

63      public OrganismCountQuery(IApplicationServerApi v3, String sessionToken, String ncbiTaxUrl, double domainThreshold) {
64  
65          this.v3 = v3;
66          this.sessionToken = sessionToken;
67          this.ncbiTaxonomyRestUrl = ncbiTaxUrl;
68          this.DOMAIN_THRESHOLD = domainThreshold;
69      }

71      /**
72       * This query executes the following steps:
73       * 1. Retrieve all taxonomy IDs currently available in OpenBis and map IDs to names
74       * 2. Get all samples of type Q_BIOLOGICAL_ENTITY
75       * 3. Remove all samples belonging to spaces that have been blacklisted
76       * 4. Group the samples by their organism, aka count all mouse samples and so on. Samples are encoded with tax ID
77       * 5. Retrieve domain and genus from NCBI for each sample
78       * 6. Count samples by domain
79       * 7. Organisms that have a large share in their respective domain(> domainThreshold) are removed from organismCountMap
80       *    and added to domainCountMap
81       * 8. Map all samples, which are on species level, with their respective genus
82       * 9. Format data and add to config
83       * @return Map of generate configs is returned: DomainName -> Count, For each domain: SpeciesName -> Count, SpeciesName-> GenusName
84       */
85      public Map<String, ChartConfig> query() {
86  
87          logger.info("Run organism count query");
88  
89          clearMaps();
90  
91          logger.info("Retrieve names of taxonomy ids.");
92          mapTaxonomyIDToName();
93  
94          logger.info("Count OpenBis samples on species basis.");
95          retrieveSamplesFromOpenBis();
96  
97          //TODO comment this back in
98          //TODO however for testing I somehow only have access to chickenfarm stuff anymore, so it has to be commented out
99          removeBlacklistedSpaces();
100 
101         countSamplesPerOrganism();
102 
103         logger.info("Map species to domain and genus");
104         setOrganismToDomainAndGenusMap();
105 
106         logger.info("Count samples on domain basis");
107         generateDomainCountMap();
108 
109         logger.info("Retrieve species with a larger share than " + DOMAIN_THRESHOLD + " in their domain.");
110         filterForLargeOrganisms();
111 
112         domainCountMap.keySet().forEach(domain -> {
113             if (!(domain.equals("Other") || domain.equals("unclassified sequences"))
114                     && Kingdoms.getList().contains(domain)) { //exclude species with large share, which were added to superkingdom resolution from being further classified
115                 logger.info("Create genus and species count map of " + domain);
116                 genusCountMaps.put(domain, new HashMap<>());
117                 speciesCountMaps.put(domain, new HashMap<>());
118             }
119         });
120 
121         generateGenusCountMap();
122         generateSpeciesCountMap();
123 
124         Map<String, ChartConfig> result = new HashMap<>();
125 
126         logger.info("Set results.");
127 
128         //Add Superkingdom to config
129         largeDomains.forEach(largeDomain -> domainCountMap.put("Other ".concat(largeDomain), domainCountMap.get(largeDomain)));
130         largeDomains.forEach(domainCountMap::remove);
131         result.put("SuperKingdom", Helpers.addPercentages( Helpers.generateChartConfig(domainCountMap, "SuperKingdom", "Sample Count by Domain", "Organisms")));
132 
133         //Add Genus maps to config
134         genusCountMaps.keySet().forEach(domain ->
135                 result.put(domain.concat("_Genus"), Helpers.addPercentages(Helpers.generateChartConfig(genusCountMaps.get(domain), domain, "Sample Count ".concat(domain), "Organisms"))));
136         //Add Species to config
137         speciesCountMaps.keySet().forEach(domain ->
138                 result.put(domain.concat("_Species"), Helpers.addPercentages(Helpers.generateChartConfig(speciesCountMaps.get(domain), domain, "", "Organisms"))));
139 
140         //Add species to genus map
141         result.put(ChartNames.Species_Genus.toString(), Helpers.generateChartConfig(organismNameGenusMap, ChartNames.Species_Genus.toString(), "", "Organisms"));
142 
143         return result;
144     }

146     private void clearMaps() {
147         domainCountMap.clear();
148         organismCountMap.clear();
149         organismDomainMap.clear();
150         organismGenusMap.clear();
151         genusCountMaps.clear();
152         speciesCountMaps.clear();
153     }

155     private void retrieveSamplesFromOpenBis() {
156 
157         SampleSearchCriteria sampleSourcesCriteria = new SampleSearchCriteria();
158         sampleSourcesCriteria.withType().withCode().thatEquals(SampleType.Q_BIOLOGICAL_ENTITY.toString());
159 
160         SampleFetchOptions fetchOptions = new SampleFetchOptions();
161         fetchOptions.withType();
162         fetchOptions.withProject();
163         fetchOptions.withSpace();
164         fetchOptions.withProperties();
165         searchResult = v3.searchSamples(sessionToken, sampleSourcesCriteria, fetchOptions);
166 
167     }

169     @Override
170     void  removeBlacklistedSpaces(){
171 
172         List<Sample> removables = new ArrayList<>();
173         searchResult.getObjects().forEach(experiment -> {
174             if (SpaceBlackList.getList().contains(experiment.getSpace().getCode())) {
175                 removables.add(experiment);
176             }
177         });
178         searchResult.getObjects().removeAll(removables);
179 
180         logger.info("Removed results from blacklisted OpenBis Spaces: " + SpaceBlackList.getList().toString());
181 
182     }

184     private void countSamplesPerOrganism() {
185         //Iterate over all search results
186         searchResult.getObjects().forEach(sample -> Helpers.addEntryToStringCountMap(organismCountMap, sample.getProperties().get(QbicPropertyType.Q_NCBI_ORGANISM.toString()), 1));
187     }

189     private void setOrganismToDomainAndGenusMap() {
190         organismCountMap.keySet().forEach(organism -> {
191             if (organism.equals("0")) { //0 = 'Other' in domain and can't be queried to NCBI
192                 organismDomainMap.put(organism, "Other");
193             } else {
194                 retrieveSuperKingdomAndGenusFromNCBI(organism);
195             }
196         });
197     }

199     //FIXME: Check with Andreas when the new XML format is implemented
200     private void retrieveSuperKingdomAndGenusFromNCBI(String organism) {
201         try (BufferedReader rd = new BufferedReader(new InputStreamReader((REST.call(ncbiTaxonomyRestUrl.concat(organism).concat("&retmode=xml")))))) {
202             String line;
203             String scientificName = "";
204             while ((line = rd.readLine()) != null) {
205                 //Retrieve Genus
206                 if (line.contains("<Rank>genus")) {
207                     organismGenusMap.put(organism, getScientificName(scientificName));
208                 }
209 
210                 //Retrieve Superkingdom
211                 if (line.contains("<Rank>superkingdom")) {
212                     organismDomainMap.put(organism, getScientificName(scientificName));
213                 }
214 
215                 //Handle unclassified samples
216                 if (line.contains("<Lineage>unclassified sequences</Lineage>")) {
217                     organismDomainMap.put(organism, "unclassified sequences");
218                 }
219 
220                 if (line.contains("<ScientificName>")) {
221                     scientificName = line.trim();
222                 }
223             }
224             TimeUnit.MILLISECONDS.sleep(340);
225         } catch (IOException|InterruptedException e) {
226             logger.error("NCBI access to retrieve names of tax ids failed: " + e.getMessage());
227         }
228     }

230     private String getScientificName(String line) {
231         //Extract Scientific name with regex
232         Pattern p = Pattern.compile("(<ScientificName>)(.*?)(<\\/ScientificName>)");
233         Matcher m = p.matcher(line);
234         m.find();
235 
236         return m.group(2);
237     }

239     private void generateDomainCountMap() {
240         organismDomainMap.keySet().forEach(o ->
241                 Helpers.addEntryToStringCountMap(domainCountMap, organismDomainMap.get(o), organismCountMap.get(o)));
242     }

244     private void filterForLargeOrganisms() {
245 
246         final Map<String, Integer> subtractionMap = new HashMap<>();
247         organismCountMap.keySet().forEach(o -> {
248             double perc = 100.0 * (double) ((int) organismCountMap.get(o)) / (double) ((int) domainCountMap.get(organismDomainMap.get(o)));
249             if (perc > DOMAIN_THRESHOLD && perc < 100.0) {
250                 largeSpecies.add(o);
251 
252                 domainCountMap.put(vocabularyMap.get(o), organismCountMap.get(o));
253                 int currcount = 0;
254                 if(subtractionMap.containsKey(organismDomainMap.get(o))) {
255                     currcount = subtractionMap.get(organismDomainMap.get(o));
256                 }
257                 subtractionMap.put(organismDomainMap.get(o), currcount + organismCountMap.get(o));
258                 largeDomains.add(organismDomainMap.get(o));
259             }
260 
261         });
262 
263         subtractionMap.keySet().forEach(key ->{
264             int currCount = (int) domainCountMap.get(key);
265             domainCountMap.put(key, currCount - subtractionMap.get(key));
266         });
267 
268     }
269 
270     private void generateGenusCountMap() {
271 
272         organismGenusMap.keySet().forEach(organism -> {
273             if (!largeSpecies.contains(organism)) {
274                 if (Kingdoms.getList().contains(organismDomainMap.get(organism))) {
275                     Helpers.addEntryToStringCountMap(genusCountMaps.get(organismDomainMap.get(organism)), organismGenusMap.get(organism), organismCountMap.get(organism));
276                 }
277             }
278         });
279     }
280 
281     private void generateSpeciesCountMap() {
282 
283         organismDomainMap.keySet().forEach(organism -> {
284             organismNameGenusMap.put(vocabularyMap.get(organism), organismGenusMap.get(organism));
285             if (!largeSpecies.contains(organism)) {
286                 if (Kingdoms.getList().contains(organismDomainMap.get(organism))) {
287                     speciesCountMaps.get(organismDomainMap.get(organism)).put(vocabularyMap.get(organism), organismCountMap.get(organism));
288                 }
289             }
290         });
291     }

293     private void mapTaxonomyIDToName() {
294         VocabularyTermFetchOptions vocabularyFetchOptions = new VocabularyTermFetchOptions();
295         vocabularyFetchOptions.withVocabulary();
296 
297         VocabularyTermSearchCriteria vocabularyTermSearchCriteria = new VocabularyTermSearchCriteria();
298         vocabularyTermSearchCriteria.withCode();
299 
300         SearchResult<VocabularyTerm> vocabularyTermSearchResult = v3.searchVocabularyTerms(sessionToken, vocabularyTermSearchCriteria, vocabularyFetchOptions);
301 
302         vocabularyTermSearchResult.getObjects().forEach(v ->
303                 vocabularyMap.put(v.getCode(), v.getLabel())
304         );
305     }
